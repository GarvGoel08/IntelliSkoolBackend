API Documentation

POST /api/user/signup
Input:
{
  "name": "John",
  "email": "john@example.com",
  "password": "123456",
  "type": "Student"
}
Output:
{
  "token": "...",
  "user": {
    "name": "John",
    "email": "john@example.com",
    "type": "Student"
  }
}

Input:
{

  "password": "123456"
}
Output:
{
  "token": "...",
  "user": {
    "name": "John",
    "email": "john@example.com",
    "type": "Student"
  }
}
GET /api/user/me

}
POST /api/classrooms
Input:
{
  "className": "Physics",
  "classDescription": "Kinematics",
  "classCode": "PHY101"
}
Output:
{
  "className": "Physics",
  "classDescription": "Kinematics",
  "classCode": "PHY101"
}
GET /api/classrooms/:id
Output:
[
  {
    "className": "Physics",
    "classDescription": "Kinematics",
    "classCode": "PHY101"
  }
]
PUT /api/classrooms/:id
Input:
{
  "className": "Updated Name"
}
Output:
{
  "className": "Updated Name",
  ...
}
DELETE /api/classrooms/:id
Output:
{ "message": "Classroom deleted successfully" }
